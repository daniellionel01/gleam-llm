import gleam/list
import gleam/option.{type Option, None, Some}

pub fn transpose(matrix: List(List(Int))) -> List(List(Int)) {
  case split(matrix, [], []) {
    Some(#([], _)) -> []
    Some(#(heads, tails)) -> [heads, ..transpose(tails)]
    None -> []
  }
}

fn split(
  rows: List(List(Int)),
  heads: List(Int),
  tails: List(List(Int)),
) -> Option(#(List(Int), List(List(Int)))) {
  case rows {
    [] -> Some(#(list.reverse(heads), list.reverse(tails)))
    [first, ..rest] ->
      case first {
        [h, ..t] -> split(rest, [h, ..heads], [t, ..tails])
        [] -> None
      }
  }
}

pub fn transpose(matrix: List(List(Int))) -> List(List(Int)) {
  case matrix {
    [] -> []
    [first_row, ..] if list.is_empty(first_row) -> []
    rows -> {
      let first_column_as_row = list.map(rows, fn(row) {
        case row {
          [head, ..] -> head
          [] -> panic as "Non-rectangular matrix encountered: row is shorter than previous rows."
        }
      })

      let remaining_matrix = list.map(rows, list.drop_left(1))

      [first_column_as_row, ..transpose(remaining_matrix)]
    }
  }
}

  let matrix = [[1, 2, 3], [4, 5, 6]]
  let transposed_matrix = transpose(matrix)

import gleam/list

pub fn transpose(matrix: List(List(Int))) -> List(List(Int)) {
  case matrix {
    [] -> []
    [[] | ..] -> []
    _ -> {
      let #(heads, tails) =
        list.fold(from: matrix, over: #([], []), with: fn(acc, row) {
          let #(h_acc, t_acc) = acc
          let [head, ..tail] = row
          #([head, ..h_acc], [tail, ..t_acc])
        })

      [
        list.reverse(heads),
        ..transpose(list.reverse(tails)),
      ]
    }
  }
}

pub fn main() {
  [[1, 2, 3], [4, 5, 6]]
  |> transpose
  |> echo
}
