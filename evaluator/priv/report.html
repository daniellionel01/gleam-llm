fn get_current_column(matrix: List(List(Int))) -> List(Int) {
  list.filter_map(matrix, fn(row) {
    case row {
      [head, .._] -> Ok(head)
      [] -> Error(Nil)
    }
  })
}

fn get_remaining_matrix(matrix: List(List(Int))) -> List(List(Int)) {
  list.map(matrix, fn(row) {
    case row {
      [_head, ..tail] -> tail
      [] -> []
    }
  })
}

pub fn transpose(matrix: List(List(Int))) -> List(List(Int)) {
  case matrix {
    [] -> []
    _ -> {
      let current_column = get_current_column(matrix)
      case current_column {
        [] -> []
        _ -> {
          let remaining_matrix = get_remaining_matrix(matrix)
          [current_column, ..transpose(remaining_matrix)]
        }
      }
    }
  }
}
