pub fn transpose(matrix: List(List(Int))) -> List(List(Int)) {
  case matrix {
    [] -> []
    [[], .._] -> [] // If the first row is empty, assume all rows are exhausted for columns
    _ -> {
      let current_column = list.map(matrix, fn(row) {
        case row {
          [head, .._] -> head
          // This case should not be reached for a rectangular matrix
          // as all rows would become empty simultaneously, triggering the base case.
          [] -> panic as "Attempted to get head of an empty row in a non-empty matrix."
        }
      })

      let remaining_matrix = list.map(matrix, fn(row) {
        case row {
          [_head, ..tail] -> tail
          [] -> [] // This row is exhausted, pass an empty list
        }
      })
      
      [current_column, ..transpose(remaining_matrix)]
    }
  }
}

fn transpose(matrix: List(List(Int))) -> List(List(Int)) {

    [] -> []
    _ -> {
      let heads = list.filter_map(matrix, list.first)
